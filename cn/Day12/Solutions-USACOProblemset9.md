    // Pythagorean Theorem 
    distSquared = (X[i]-X[j])^2 + (Y[i]-Y[j])^2 
    sumOfR = R[i] + R[j] 
    sumOfRSquared = sumOfR * sumOfR 
    if sumOfRSquared > distSquared: 
        return true 

    return false
